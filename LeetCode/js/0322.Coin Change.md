```javascript
var coinChange = function(coins, amount) {
    coins.sort((a, b) => a - b);
    var cell = new Array(amount+1).fill(-1);
    cell[0] = 0;
    for(var coin of coins){
        for(var i = 1; i < amount+1; i++){
            if(i >= coin){
                if(cell[i] == -1){
                    cell[i] = cell[i - coin] + 1;
                }else {
                    cell[i] = Math.min((cell[i - coin] + 1), cell[i]);
                }
            }
        }
    }
    console.log(cell);
    return cell[amount];
};
```
